            throw URLError(.cannotDecodeContentData)
        }
        return text

func sendMessage(text: String, images: [UIImage]) {
    let trimmedText = text.trimmingCharacters(in: .whitespacesAndNewlines)
    // Require either text or images
    guard !trimmedText.isEmpty || !images.isEmpty else {
        return
    }
    isLoading = true
    // Create user message
    let userMessage = Message(
        id: UUID(),
        content: trimmedText.isEmpty ? "[Image]" : trimmedText,
        isUser: true,
        timestamp: Date(),
        image: images.first  // Store first image for display
    )
    messages.append(userMessage)
    Task {
        do {
            // Send to backend (handles S3 upload internally)
            let responseText = try await service.sendMessage(
                text: trimmedText,
                images: images,
                conversationHistory: messages
            )
            await MainActor.run {
                let aiMessage = Message(
                    id: UUID(),
                    content: responseText,
                    isUser: false,
                    timestamp: Date()
                )
                messages.append(aiMessage)
                isLoading = false
            }
        } catch {
            await MainActor.run {
                errorMessage = error.localizedDescription
                showError = true
                isLoading = false
            }
        }
    }
}
