Script 1: generate-upload-url Lambda
This generates presigned URLs for iOS to upload images directly to S3.
javascript
import { S3Client, PutObjectCommand } from "@aws-sdk/client-s3";
import { getSignedUrl } from "@aws-sdk/s3-request-presigner";
const s3Client = new S3Client({ region: "ap-southeast-1" });  // Your region
const BUCKET_NAME = "your-app-chat-images";  // Your bucket name
export const handler = async (event) => {
    const headers = {
        "Access-Control-Allow-Origin": "*",
        "Access-Control-Allow-Headers": "Content-Type",
        "Content-Type": "application/json"
    };
    // Handle CORS preflight
    if (event.httpMethod === "OPTIONS") {
        return { statusCode: 200, headers, body: "" };
    }
    try {
        // Generate unique filename
        const timestamp = Date.now();
        const randomId = Math.random().toString(36).substring(2, 8);
        const imageKey = `uploads/${timestamp}-${randomId}.jpg`;
        // Create presigned URL (valid for 5 minutes)
        const command = new PutObjectCommand({
            Bucket: BUCKET_NAME,
            Key: imageKey,
            ContentType: "image/jpeg"
        });
        const uploadUrl = await getSignedUrl(s3Client, command, { expiresIn: 300 });
        return {
            statusCode: 200,
            headers,
            body: JSON.stringify({
                uploadUrl: uploadUrl,    // iOS uploads to this URL
                imageKey: imageKey       // iOS sends this key to chat Lambda
            })
        };
    } catch (error) {
        console.error("Error generating presigned URL:", error);
        return {
            statusCode: 500,
            headers,
            body: JSON.stringify({ error: error.message })
        };
    }
};
