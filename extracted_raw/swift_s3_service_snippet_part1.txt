struct S3UploadService {
    struct PresignedURLResponse: Codable {
        let uploadUrl: String
        let imageKey: String
    }
    // Step 1: Get presigned URL from backend
    static func getPresignedURL() async throws -> PresignedURLResponse {
        guard let url = URL(string: AppConfig.uploadURLEndpoint) else {
            throw URLError(.badURL)
        }
        var request = URLRequest(url: url)
        request.httpMethod = "GET"
        request.timeoutInterval = 30
        let (data, response) = try await URLSession.shared.data(for: request)
        guard let httpResponse = response as? HTTPURLResponse,
              httpResponse.statusCode == 200 else {
            throw URLError(.badServerResponse)
        }
        return try JSONDecoder().decode(PresignedURLResponse.self, from: data)
    }
    // Step 2: Upload image to S3 using presigned URL
    static func uploadImage(_ image: UIImage, to presignedURL: String) async throws {
        guard let url = URL(string: presignedURL) else {
            throw URLError(.badURL)
        }
        // Compress image
        guard let imageData = image.jpegData(compressionQuality: 0.6) else {
            throw URLError(.cannotDecodeContentData)
        }
        var request = URLRequest(url: url)
        request.httpMethod = "PUT"
        request.setValue("image/jpeg", forHTTPHeaderField: "Content-Type")
        request.httpBody = imageData
        request.timeoutInterval = 60
        let (_, response) = try await URLSession.shared.data(for: request)
        guard let httpResponse = response as? HTTPURLResponse,
              (200...299).contains(httpResponse.statusCode) else {
            throw URLError(.badServerResponse)
        }
