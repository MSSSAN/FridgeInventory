const { DynamoDBClient } = require("@aws-sdk/client-dynamodb");
const { DynamoDBDocumentClient, PutCommand, QueryCommand, UpdateCommand } = require("@aws-sdk/lib-dynamodb");
const { S3Client, GetObjectCommand } = require("@aws-sdk/client-s3");
const dynamoClient = new DynamoDBClient({ region: "ap-southeast-1" });
const docClient = DynamoDBDocumentClient.from(dynamoClient);
const s3Client = new S3Client({ region: "ap-southeast-1" });
const TABLE_NAME = "ChatHistory";
const BUCKET_NAME = "chat-images-store-bucket";
const OPENAI_API_KEY = process.env.OPENAI_API_KEY;
const OPENAI_API_URL = "https://api.openai.com/v1/chat/completions";
exports.handler = async (event) => {
    console.log("Worker started:", JSON.stringify(event));
    try {
        const { sessionID, userText, imageKeys, requestId, userTimestamp } = event;
        // Fetch conversation history
        const historyCommand = new QueryCommand({
            TableName: TABLE_NAME,
            KeyConditionExpression: "sessionID = :s",
            ExpressionAttributeValues: { ":s": sessionID },
            Limit: 20,
            ScanIndexForward: true
        });
        const historyResult = await docClient.send(historyCommand);
        const previousMessages = historyResult.Items
            .filter(item => item.status === "completed")  // Only completed messages
            .map(item => ({
                role: item.role,
                content: item.content
            }));
        // Build current message with images from S3
        let currentContent = [];
        if (userText) {
            currentContent.push({ type: "text", text: userText });
        }
        if (imageKeys && imageKeys.length > 0) {
            for (const imageKey of imageKeys) {
                const imageBase64 = await getImageFromS3(imageKey);
                currentContent.push({
                    type: "image_url",
                    image_url: {
                        url: `data:image/jpeg;base64,${imageBase64}`
                    }
                });
            }
        }
        const currentUserMessage = {
            role: "user",
            content: currentContent.length === 1 && currentContent[0].type === "text"
                ? currentContent[0].text
                : currentContent
        };
        const allMessages = [...previousMessages, currentUserMessage];
        // Call OpenAI
        const openaiResponse = await fetch(OPENAI_API_URL, {
            method: "POST",
            headers: {
                "Content-Type": "application/json",
                "Authorization": `Bearer ${OPENAI_API_KEY}`
            },
            body: JSON.stringify({
                model: "gpt-4o-mini",
                messages: allMessages,
                max_tokens: 4096
            })
        });
        const openaiData = await openaiResponse.json();
        if (!openaiResponse.ok) {
            throw new Error(openaiData.error?.message || "OpenAI API error");
        }
        const aiResponseText = openaiData.choices[0].message.content;
        // Update user message status to completed
        await docClient.send(new UpdateCommand({
            TableName: TABLE_NAME,
            Key: {
                sessionID: sessionID,
                timestamp: userTimestamp
            },
            UpdateExpression: "SET #status = :completed",
            ExpressionAttributeNames: {
                "#status": "status"
            },
            ExpressionAttributeValues: {
                ":completed": "completed"
            }
        }));
        // Save AI response
        await docClient.send(new PutCommand({
            TableName: TABLE_NAME,
            Item: {
                sessionID: sessionID,
                timestamp: userTimestamp + 1,
                role: "assistant",
                content: aiResponseText,
                status: "completed"
            }
        }));
        console.log("Worker completed successfully");
        return { success: true };
    } catch (error) {
        console.error("Worker error:", error);
        // Mark as failed in DynamoDB
        try {
            await docClient.send(new UpdateCommand({
                TableName: TABLE_NAME,
                Key: {
                    sessionID: event.sessionID,
                    timestamp: event.userTimestamp
                },
                UpdateExpression: "SET #status = :failed, errorMessage = :msg",
                ExpressionAttributeNames: {
                    "#status": "status"
                },
                ExpressionAttributeValues: {
                    ":failed": "failed",
                    ":msg": error.message
                }
            }));
        } catch (updateError) {
            console.error("Failed to update error status:", updateError);
        }
        throw error;
    }
};
async function getImageFromS3(imageKey) {
    const command = new GetObjectCommand({
        Bucket: BUCKET_NAME,
        Key: imageKey
    });
    const response = await s3Client.send(command);
    const buffer = await streamToBuffer(response.Body);
    return buffer.toString("base64");
}
async function streamToBuffer(stream) {
    const chunks = [];
    for await (const chunk of stream) {
        chunks.push(chunk);
    }
    return Buffer.concat(chunks);
}
